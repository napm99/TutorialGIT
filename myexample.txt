 public static void uploadMoneyOrder() throws IOException {
        DateTimeFormatter dtf = DateTimeFormatter.ofPattern("yyyyMMddHHmm");
        LocalDateTime now = LocalDateTime.now();

        String tempFile = "//MIRSRVQAAPP01.QAJPAY.COM//MoneyOrders//Deposits//" + "DepositV2_Automated_" + dtf.format(now)+".txt";
        File oldFile = new File("//MIRSRVQAAPP01.QAJPAY.COM//MoneyOrders//Archive//01102023DepositV2_jb_01101.txt");
        File newFile = new File(tempFile);

        try {

            FileWriter fw = new FileWriter(tempFile, true);
            BufferedWriter bw = new BufferedWriter(fw);
            PrintWriter pw = new PrintWriter(bw);
            scn = new Scanner(new File("//MIRSRVQAAPP01.QAJPAY.COM//MoneyOrders//Archive//01102023DepositV2_jb_01101.txt"));
            BufferedReader reader = new BufferedReader(new FileReader("//MIRSRVQAAPP01.QAJPAY.COM//MoneyOrders//Archive//01102023DepositV2_jb_01101.txt"));


            String line = reader.readLine();
            pw.println("HEADER");


            while((line = reader.readLine()) != null){

                String changeMoId = dtf.format(now);
                String digitsToChange = line.substring(0,12);
                String endLine = line.substring(0,6);
                if (endLine.equals("FOOTER")){
                    pw.println("FOOTER               0270900");
                    break;
                }
                pw.println(line.replace(digitsToChange,changeMoId));



            }
            scn.close();
            pw.flush();
            pw.close();
            File dump = new File("//MIRSRVQAAPP01.QAJPAY.COM//MoneyOrders//Archive//01102023DepositV2_jb_01101.txt");
            newFile.renameTo(newFile);
        } catch (Exception e) {
            e.printStackTrace();
        }
    }
